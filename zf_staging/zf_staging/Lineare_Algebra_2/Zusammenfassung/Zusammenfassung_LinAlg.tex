    \section{Lineare Unabhängigkeit}
    
    Definition
    
    $$\sum_{i=1}^n\lambda_i v_i = 0 \implies \lambda_i = 0 \quad für \quad i=1,...,n$$
    
    Vektoren sind linear ubahängig, falls die summe der Vektoren nur dann null ist, wenn die Faktoren 0 sind. Sind aber nicht alle Fakoten 0 so sind die Vekotren linear abhängig.
    
    $$\lambda_1 \vec{a}_1 + \lambda_2 \vec{a}_2 + ... + \lambda_k \vec{a}_k = \vec{0}$$
    
    Vektoren sind linear unabhängig, falls ein Vektor nicht durch eine linearkombinationder anderen dargestellt werden kann. \\
    
    1. Es lässt sich zeigen, dass es in der Ebene maximal zwei lineare unabhänige Vektoren gibt (daher stammt auch die Bezeihnung "2-dimensionaler" Raum für die Ebene), mehr als zwei Vektoren sind immer linear abhängig. Gilt auch für K=3, einfach dass es hier mehr wie 3 Vekoren sind, die zur linearen Abhänigkeit führen. \\
    
    2. Die k Vektoren sind linear Abhängig, wenn sie den Nullvketor oder kolineare Vektoren enthalten oder wenn mindestens einer der Vektoren als Linearkombination der übrigen darstellbar ist. \\
    
    3. Bei jeder Menge von Vektoren lässt sich auch durch das Lösen eines homogenen LGS bestimmen ob sie linear abhängig ist oder nicht. Falls der Rang (Anzahl Pivot) gleich der Anzahl Spalten ist, sind sie linear unabhängig. Ist der Rang ungleich der Anzahl Spalten oder gibt es unendlich viele lösungen sind sie linear abhängig. \\
    
    4. In vielen einfachen fällen "sieht" man es oder kann durch einfaches herumprobieren feststellen ob eine Menge von Vektoren linear abhängig ist oder nicht. Das geht ofet schneller als zu gaussen.
    
    \subsection*{1. Linear unabhängig}
    
    $\vec{v}_1 = \begin{pmatrix}1\\0\end{pmatrix}$, $\vec{v}_2 = \begin{pmatrix}0\\1\end{pmatrix}$
    
    \[
    \left(\begin{array}{cc|c} 1 & 0 & 0 \\ 0 & 1 & 0 \end{array}\right)
    \xrightarrow{\text{bereits in Stufenform}}
    \left(\begin{array}{cc|c} 1 & 0 & 0 \\ 0 & 1 & 0 \end{array}\right)
    \]
    
    Rang $= 2 =$ Anzahl Vektoren $\Rightarrow \lambda_1 = 0,\; \lambda_2 = 0$
    $\Rightarrow$ linear \textbf{unabhängig}.
    
    \subsection*{2. Linear abhängig}
    
    $\vec{v}_1 = \begin{pmatrix}1\\2\end{pmatrix}$, $\vec{v}_2 = \begin{pmatrix}2\\4\end{pmatrix}$
    
    \[
    \left(\begin{array}{cc|c} 1 & 2 & 0 \\ 2 & 4 & 0 \end{array}\right)
    \xrightarrow{Z_2 \,-=\, 2 \cdot Z_1}
    \left(\begin{array}{cc|c} 1 & 2 & 0 \\ 0 & 0 & 0 \end{array}\right)
    \]
    
    Rang $= 1 <$ Anzahl Vektoren $\Rightarrow \lambda_2 = t,\; \lambda_1 = -2t$
    $\Rightarrow$ linear \textbf{abhängig}.
    
    \subsection{Lineare Hülle}
    
    Seien $V$ ein $\mathbb{K}$-Vektorraum und $S \subset V$ eine Teilmenge von $V$.
    
    \begin{enumerate}
    	\item[i)] Seien $v_1, \ldots, v_n \in V,\ n \in \mathbb{N}$ beliebige Vektoren in $V$. 
    	Die Menge aller Linearkombinationen von $v_1, \ldots, v_n$
    	\[
    	\text{span}(v_1, \ldots, v_n) := \left\{ v \in V \;\middle|\; v = \sum_{i=1}^{n} \lambda_i v_i,\ \lambda_i \in \mathbb{K} \right\}
    	\]
    	heisst die \textbf{lineare Hülle} von $v_1, \ldots, v_n$.
    	
    	\item[ii)] Sei $S \subset V$ eine nicht leere Teilmenge von $V$. 
    	Dann ist die \textbf{lineare Hülle} von $S$ definiert durch
    	\[
    	\text{span}(S) := \left\{ v \in V \;\middle|\; v = \sum_{i=1}^{n} \lambda_i v_i,\ v_i \in S,\ \lambda_i \in \mathbb{K},\ n \in \mathbb{N} \right\}
    	\]
    	
    	Zusätzlich legen wir noch fest:
    	\[
    	\text{span}(\emptyset) := \{0\},
    	\]
    \end{enumerate}
    
    \textbf{eine Lineare hülle muss nich zwingend linear unabhängig sein, nur wenn wir sie als basis betrachten}
    
    \subsubsection{Erzeugendensysteme}
    
    Seien $V$ ein $\mathbb{K}$-Vektorraum und $S \subset V$ eine Teilmenge von $V$. 
    $S$ heisst \textbf{Erzeugendensystem} von $V$, falls
    \[
    \text{span}(S) = V,
    \]
    d.h. $S$ spannt den gesamten Vektorraum $V$ auf.

    \section{Norm}
    
    \subsection{Vektornormen}
    
    \subsubsection{euklidische Norm}
    $$\| \vec{v} \|_2  = \sqrt{v_1^2 + v_2^2 + \cdots + v_n^2}$$
    
    $$\left\| \begin{pmatrix} 3 \\ 4 \end{pmatrix} \right\| = \sqrt{3^2 + 4^2} = \sqrt{25} = 5$$
    
    \subsubsection{p-Norm}
    
    $$\|\vec{v}\|_p = \left( \sum_{i=1}^{n} |v_i|^p \right)^{\frac{1}{p}}$$
    
    % 1-Norm (Betragssummennorm)
    $$\|\vec{v}\|_1 = |v_1| + |v_2| + \cdots + |v_n|$$
    
    % 2-Norm (Euklidische Norm)
    $$\|\vec{v}\|_2 = \sqrt{v_1^2 + v_2^2 + \cdots + v_n^2}$$
    
    % p-Norm allgemein
    $$\|\vec{v}\|_p = \left( |v_1|^p + |v_2|^p + \cdots + |v_n|^p \right)^{\frac{1}{p}}$$
    
    \subsubsection{Maximumsnorem}
    
    % Unendlich-Norm (Maximumsnorm)
    $$\|\vec{v}\|_\infty = \max_{i} |v_i|$$
    
    $$\left\| \begin{pmatrix} 3 \\ -7 \\ 2 \end{pmatrix} \right\|_\infty$$
    $$= \max(|3|, |-7|, |2|) $$
    $$= \max(3, 7, 2) $$
    $$= 7$$
    
    \subsection{Matrixnorm}
    
    \textbf{Beispielmatrix:}
    $$A = \begin{pmatrix} 1 & -2 \\ 3 & 4 \end{pmatrix}$$
    
    \subsubsection{Frobenius-Norm}
    $$\|A\|_F := \sqrt{\sum_{i=1}^{m} \sum_{j=1}^{n} |a_{ij}|^2}$$
    Beispiel:
    $$\|A\|_F = \sqrt{|1|^2 + |-2|^2 + |3|^2 + |4|^2} 
    = \sqrt{1 + 4 + 9 + 16} 
    = \sqrt{30}$$
    
    \subsubsection{Zeilensummennorm}
    $$\|A\|_\infty := \max_{1 \leq i \leq m} \sum_{j=1}^{n} |a_{ij}|$$
    Beispiel:
    $$Z_1 = |1| + |-2| = 3$$
    $$Z_2 = |3| + |4|  = 7$$
    $$\|A\|_\infty = \max(3, 7) = 7$$
    
    \subsubsection{Spaltensummennorm}
    $$\|A\|_1 := \max_{1 \leq j \leq n} \sum_{i=1}^{m} |a_{ij}|$$
    Beipsiel:
    $$S_1 = |1| + |3|  = 4$$
    $$S_2 = |-2| + |4| = 6$$
    $$\|A\|_1 = \max(4, 6) = 6$$
    
    \subsection{Funktionsnorm}
    
    Die $L^p$
    Lp-Norm verallgemeinert die Vektornormen auf
    Funktionen – statt Summe über Einträge steht ein Integral.
    
    $$L^p(\mathbb{D}) := \left\{ f : \mathbb{D} \to \mathbb{K} \;\middle|\; \int_{\mathbb{D}} |f(x)|^p \, dx < \infty \right\}$$
    
    $$\|f\|_{L^p} := \left( \int_{\mathbb{D}} |f(x)|^p \, dx \right)^{1/p}$$
    
    Beispiel:
    
    % Sei f(x) = x, D = [0,1], p = 2
    
    $$\|f\|_{L^2} = \left( \int_0^1 |x|^2 \, dx \right)^{1/2}$$
    $$= \left( \int_0^1 x^2 \, dx \right)^{1/2}$$
    $$= \left[ \frac{x^3}{3} \right]_0^1 )^{1/2}$$
    $$= \left( \frac{1}{3} \right)^{1/2}$$
    $$= \frac{1}{\sqrt{3}}$$
    
    \section{Skalarprodukt}
    
    \subsection{Standardskalarprodukt in $R^n$}
    $$\langle x, y \rangle := \sum_{i=1}^{n} x_i y_i, \quad x, y \in \mathbb{R}^n$$
    Beispiel:
    % R^n: x = (1,2), y = (3,4)
    $$\langle x, y \rangle = 1 \cdot 3 + 2 \cdot 4 = 11$$
    
    \subsubsection{Standardskalarprodukt in $C^n$}
    $$\langle w, z \rangle := \sum_{i=1}^{n} \overline{w_i} z_i, \quad w, z \in \mathbb{C}^n$$
    
    \subsection{Skalarprodukt für reelle Matrizen}
    $$\langle A, B \rangle := \text{tr}(A^T B), \quad A, B \in \mathbb{R}^{m \times n}$$
    
    Beispiel:
    
    % Matrizen: A = B = \begin{pmatrix} 1 & 2 \\ 3 & 4 \end{pmatrix}
    $$A^T B = \begin{pmatrix} 1 & 3 \\ 2 & 4 \end{pmatrix} \begin{pmatrix} 1 & 2 \\ 3 & 4 \end{pmatrix}
    = \begin{pmatrix} 10 & 14 \\ 14 & 20 \end{pmatrix}
    \langle A, B \rangle = \text{tr}(A^T B) = 10 + 20 = 30$$
    
    \subsection{Skalarprodukt für komplexe Matrizen}
    $$\langle A, B \rangle := \text{tr}(A^* B), \quad A, B \in \mathbb{C}^{m \times n}$$
    
    \subsubsection{Skalarprodukt auf $L^2$}
    $$\langle f, g \rangle_{L^2} := \int_{\mathbb{D}} \overline{f(x)} g(x) \, dx$$
    
    Beispiel:
    
    % L^2: f(x) = x, g(x) = x^2, D = [0,1]
    $$\langle f, g \rangle_{L^2} = \int_0^1 x \cdot x^2 \, dx = \int_0^1 x^3 \, dx = \frac{1}{4}$$
    
    \subsection{Induzierte Norm}
    
    % Erklärsatz
    \textbf{Induzierte Norm:} \text{ Jedes Skalarprodukt erzeugt automatisch eine Norm via } 
    $\|v\| := \sqrt{\langle v, v \rangle} $ \\
    \text{ – man setzt einfach beide Argumente gleich und zieht die Wurzel.}
    
    \text{Nicht jede Norm kommt von einem Skalarprodukt, aber jedes Skalarprodukt liefert immer eine Norm.}
    
    % Allgemein: induzierte Norm
    $$\|v\| := \sqrt{\langle v, v \rangle}$$
    
    % R^n
    $$\|x\| = \sqrt{\langle x, x \rangle} = \sqrt{\sum_{i=1}^{n} x_i^2}$$
    
    % C^n
    $$\|w\| = \sqrt{\langle w, w \rangle} = \sqrt{\sum_{i=1}^{n} \overline{w_i} w_i} = \sqrt{\sum_{i=1}^{n} |w_i|^2}$$
    
    % Matrizen R^{m x n}  (= Frobenius-Norm!)
    $$\|A\|_F = \sqrt{\langle A, A \rangle} = \sqrt{\text{tr}(A^T A)}$$
    
    % Matrizen C^{m x n}
    $$\|A\|_F = \sqrt{\langle A, A \rangle} = \sqrt{\text{tr}(A^* A)}$$
    
    % L^2
    $$\|f\|_{L^2} = \sqrt{\langle f, f \rangle} = \sqrt{\int_{\mathbb{D}} |f(x)|^2 \, dx}$$
    
    Beispiele: 
    
    % R^n: x = (1,2)
    $$\|x\| = \sqrt{1^2 + 2^2} = \sqrt{5}$$
    
    % Matrizen: A = \begin{pmatrix} 1 & 2 \\ 3 & 4 \end{pmatrix}
    $$\text{tr}(A^T A) = \text{tr}\begin{pmatrix} 10 & 14 \\ 14 & 20 \end{pmatrix} = 30
    \|A\|_F = \sqrt{30}$$
    
    % L^2: f(x) = x, D = [0,1]
    $$\|f\|_{L^2} = \sqrt{\int_0^1 x^2 \, dx} = \sqrt{\frac{1}{3}} = \frac{1}{\sqrt{3}}$$
