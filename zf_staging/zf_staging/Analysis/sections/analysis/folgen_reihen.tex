	\subsection{Folgen}
	
	Eine Folge ist eine Abbildung
	\[
	(a_n)_{n\in\mathbb{N}}, \quad a_n \in \mathbb{R}.
	\]
	
	Die Folge \textbf{konvergiert}, falls
	\[
	\lim_{n\to\infty} a_n = L \in \mathbb{R}.
	\]
	
	\subsubsection{Wichtige Folgentypen}
	
	\textbf{Arithmetische Folge}
	
	\textbf{Definition}
	\[
	a_n = a_1 + (n-1)d
	\]
	
	\begin{itemize}
		\item $a_1$: Startwert
		\item $d$: konstante Differenz
	\end{itemize}
	
	\textbf{Grenzwert}
	\[
	\lim_{n\to\infty} a_n =
	\begin{cases}
		+\infty & d>0, \\
		-\infty & d<0, \\
		a_1 & d=0.
	\end{cases}
	\]
	
	Arithmetische Folgen konvergieren nur im trivialen Fall $d=0$.
	
	\textbf{Geometrische Folge}
	
	\textbf{Definition}
	\[
	a_n = a_1 q^{\,n-1}
	\]
	
	\begin{itemize}
		\item $a_1$: Startwert
		\item $q$: Quotient
	\end{itemize}
	
	\textbf{Grenzwert}
	\[
	\lim_{n\to\infty} a_n =
	\begin{cases}
		0 & |q|<1, \\
		a_1 & q=1, \\
		\text{existiert nicht} & q=-1, \\
		\pm\infty & |q|>1.
	\end{cases}
	\]
	
	\textbf{Potenzfolgen}
	\[
	a_n = n^p
	\]
	
	\begin{itemize}
		\item $p>0$: divergiert $(\to \infty)$
		\item $p=0$: konstant
		\item $p<0$: konvergiert gegen $0$
	\end{itemize}
	
	\textbf{Exponentialfolgen}
	\[
	a_n = q^n
	\]
	
	\begin{itemize}
		\item $|q|<1$: $a_n \to 0$
		\item $q=1$: $a_n = 1$
		\item $q=-1$: divergent
		\item $|q|>1$: divergent
	\end{itemize}
	
	\textbf{Quotienten von Polynomen}
	\[
	a_n = \frac{n^k}{n^m}
	\]
	
	Vergleich der höchsten Potenzen:
	\begin{itemize}
		\item $k<m$: Grenzwert $0$
		\item $k=m$: Quotient der Leitkoeffizienten
		\item $k>m$: divergent
	\end{itemize}
	
	\subsection{Reihen – Grundlagen}
	
	Eine Reihe ist die Summe einer Folge:
	\[
	\sum_{n=1}^{\infty} a_n.
	\]
	
	(ansonsten divergiert die Reihe sicher).
	
	\textbf{Merksatz:}  
	Ist $\displaystyle \sum_{n=1}^{\infty} a_n$ konvergent, so gilt
	\[
	a_n \xrightarrow[n\to\infty]{} 0.
	\]
	Gilt $a_n \not\to 0$, so ist die Reihe divergent. Damit diese Summe endlich bleiben kann, müssen die einzelnen Summanden immer kleiner werden und gegen 0 gehen.
	
	
	\subsection{Wichtige Reihentypen}
	
	\textbf{Arithmetische Reihe (endliche Summe)}
	
	\[
	\sum_{k=1}^{n} a_k
	\quad \text{mit} \quad
	a_k = a_1 + (k-1)d
	\]
	
	\textbf{Summenformel}
	\[
	S_n = \frac{n}{2}\bigl(2a_1 + (n-1)d\bigr)
	\]
	
	oder äquivalent:
	\[
	S_n = \frac{n}{2}(a_1 + a_n)
	\]
	
	\textbf{Hinweis:}
	Die unendliche arithmetische Reihe
	\[
	\sum_{k=1}^{\infty} a_k
	\]
	divergiert immer (außer im trivialen Fall $a_1 = d = 0$).
	
	\paragraph{1. Gausstrick}
	
	\subsubsection{Geometrische Reihe}
	\[
	\sum_{n=0}^{\infty} a q^n
	\]
	
	\textbf{Konvergenz:}
	\[
	|q| < 1
	\]
	
	\textbf{Summe:}
	\[
	S = \frac{a}{1-q}
	\]
	
	\paragraph{2. Gausstrick}
	
	\subsubsection{Harmonische Reihe}
	\[
	\sum_{n=1}^{\infty} \frac{1}{n}
	\]
	Divergent.
	
	\subsubsection{Potenzreihe}
	Beispiel:
	$$s_n=\sum_{k=1}^{n}k^2$$
	
	Für diese Reihen gilt die Summenformel:
	
	$$s_n=\frac{n(n+1)(2n+1)}{6}$$
	
	Für andere Potenzen gelten andere Summenformel, diese kann man herleiten mitthilf des Bernoulli-Tricks:
	
	\paragraph{Bernoulli-Trick}
	
	Mit dem \textbf{Bernoulli-Trick} lässt sich die Summenformel für jede Potenzreihe der Form
	\[
	\sum_{k=1}^n k^p
	\]
	bestimmen.
	
	Dazu betrachtet man die Differenz
	\[
	(k+1)^{p+1} - k^{p+1},
	\]
	da diese beim Summieren eine Teleskopsumme ergibt.
	
	Mit dem binomischen Lehrsatz wird die Differenz ausmultipliziert. Dadurch entsteht ein Polynom, das den Term \(k^p\) enthält.
	
	Nach dem Summieren über \(k\) werden konstante Faktoren ausgeklammert. Die Summe von \(k^p\) wird mit \(s_n\) bezeichnet.
	
	Da die linke Seite teleskopiert, gilt:
	\[
	\sum_{k=1}^n \big((k+1)^{p+1} - k^{p+1}\big) = (n+1)^{p+1}-1.
	\]
	
	Alle übrigen Summen mit niedrigeren Potenzen werden durch bekannte Summenformeln ersetzt. Danach löst man die Gleichung nach \(s_n\) auf.
	
	\subparagraph{Beispiel: \(\sum_{k=1}^n k^3\)}\mbox{}\\
	
	Sei
	\[
	s_n := \sum_{k=1}^n k^3.
	\]
	
	Man betrachtet:
	\[
	(k+1)^4 - k^4.
	\]
	
	Mit dem binomischen Lehrsatz:
	\[
	(k+1)^4 - k^4 = 4k^3 + 6k^2 + 4k + 1.
	\]
	
	Summiert über \(k=1\) bis \(n\):
	\[
	\sum_{k=1}^n \big((k+1)^4 - k^4\big)
	=
	\sum_{k=1}^n (4k^3 + 6k^2 + 4k + 1).
	\]
	
	Linke Seite teleskopiert:
	\[
	(n+1)^4 - 1.
	\]
	
	Rechte Seite:
	\[
	4s_n
	+ 6\sum_{k=1}^n k^2
	+ 4\sum_{k=1}^n k
	+ n.
	\]
	
	Mit bekannten Summen:
	\[
	\sum_{k=1}^n k^2 = \frac{n(n+1)(2n+1)}{6}, \quad
	\sum_{k=1}^n k = \frac{n(n+1)}{2}.
	\]
	
	Es folgt:
	\[
	4s_n = n^2(n+1)^2.
	\]
	
	Damit:
	\[
	\boxed{
		\sum_{k=1}^n k^3 = \left(\frac{n(n+1)}{2}\right)^2
	}
	\]
	
	\subsubsection{p-Reihe}
	\[
	\sum_{n=1}^{\infty} \frac{1}{n^p}
	\]
	
	\begin{itemize}
		\item $p>1$: konvergent
		\item $p \le 1$: divergent
	\end{itemize}
	
	\subsubsection{Alternierende Reihen}
	\[
	\sum (-1)^n a_n
	\]
	
	\textbf{Leibniz-Kriterium:}  
	Die Reihe konvergiert, falls:
	\begin{enumerate}
		\item $a_n \ge 0$
		\item $(a_n)$ monoton fallend
		\item $a_n \to 0$
	\end{enumerate}
	
	\subsubsection{Teleskopsummen}
	Beispiel: 
	$$s_n=\sum_{k=1}^{n}f(k+1)-f(k)$$
	
	Durch Termauschlöschung gilt:
	$$s_n=f(n+1)-f(1)$$
	
	\subsection*{4. Konvergenzkriterien}
	
	\textbf{Majorantenkriterium}
	
	Falls
	\[
	0 \le a_n \le b_n \quad \text{und} \quad \sum b_n \text{ konvergiert},
	\]
	dann konvergiert auch $\sum a_n$.
	
	\textbf{Minorantenkriterium}
	
	Falls
	\[
	a_n \ge b_n \ge 0 \quad \text{und} \quad \sum b_n \text{ divergiert},
	\]
	dann divergiert auch $\sum a_n$.
	
	\textbf{Quotientenkriterium}
	\[
	L = \lim_{n\to\infty} \left|\frac{a_{n+1}}{a_n}\right|
	\]
	
	\begin{itemize}
		\item $L<1$: konvergent
		\item $L>1$: divergent
		\item $L=1$: keine Aussage
	\end{itemize}
	
	\subsection*{5. Prüfungsstrategie (Kurzfassung)}
	
	\begin{enumerate}
		\item Prüfen: $a_n \to 0$?
		\item Geometrische Reihe erkennen?
		\item Vergleich mit p-Reihe
		\item Alternierend $\Rightarrow$ Leibniz
		\item Quotientenkriterium anwenden
	\end{enumerate}
