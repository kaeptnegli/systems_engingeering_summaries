\subsection{Eigenschaften von Funktionen}

\subsubsection{Symterieverhalten}

\paragraph{Punktsymetire - gerade Funktion}

$$f(-x)=f(x)$$

Beispiel:

$$f(x) = cos(x)$$ 
$$f(x) = x²$$

\paragraph{Achensymetire - ungerade Funktion}

$$f(-x)=-f(x)$$

Beispiel: 

$$f(x) = sin(x)$$ 
$$f(x) = x³$$

\subsubsection{Monotonie}

\textbf{Definition:} $x_1$ und $x_2$ seien zwei belibige Werte aus dem Definitionsbreich $\mathbb{D}$ eienr Funktion $y = f(x)$, die der Bedingung $x_1 < x_2$ genügt. Dann heisst die Funktion:
\\
\\
\textit{monoton wachsend, falls} $f(x_1) \le f(x_2)$ 
\\
\textit{streng monoton wachsend, falls} $f(x_1) < f(x_2)$
\\
\textit{monoton fallend, falls} $f(x_1) \ge f(x_2)$ 
\\
\textit{streng monoton fallend, falls} $f(x_1) > f(x_2)$

\subsubsection{Periodizität}

\textbf{Definition}: Eine Funktion $y=f(x)$ heisst periodisch mit der Periode $p$, wenn mit jedem $x \in \mathbb{D}$ auch $x \pm p$ zum definitionsbrereich der Funktion gehärt und
$$f(x \pm p) = f(x)$$
ist.

\begin{enumerate}
	\item mit der Periode $p$ ist auch $\pm k * p$ eine Periode der Funktion $k \in \mathbb{N}_0$
	\item Die kleinste positive Periode $p$ heisst auch primitive Periode.
\end{enumerate}
